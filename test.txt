  <h1>ドラクエ風クイズRPG</h1>
  <div id="status">プレイヤーのHP: <span id="hp">100</span> / レベル: <span id="level">1</span></div>
  <canvas id="gameCanvas" width="600" height="400"></canvas>
  
  <div id="battle-screen">
    <p id="battle-text">モンスターがあらわれた！</p>
    <button onclick="startBattle()">戦う</button>
  </div>

  <script>
    let player = { x: 300, y: 200, hp: 100, level: 1, steps: 0 };
    let ctx = document.getElementById("gameCanvas").getContext("2d");
    let inBattle = false;

    function drawPlayer() {
      ctx.clearRect(0, 0, 600, 400);
      ctx.fillStyle = "red";
      ctx.fillRect(player.x - 10, player.y - 10, 20, 20);
    }

    document.getElementById("gameCanvas").addEventListener("mousedown", startDrag);
    function startDrag(event) {
      if (inBattle) return;
      let offsetX = event.clientX - player.x;
      let offsetY = event.clientY - player.y;

      function onMouseMove(event) {
        player.x = event.clientX - offsetX;
        player.y = event.clientY - offsetY;
        player.steps++;
        drawPlayer();

        if (player.steps % Math.floor(Math.random() * 20 + 10) === 0) {
          startEncounter();
        }
      }

      function onMouseUp() {
        document.removeEventListener("mousemove", onMouseMove);
        document.removeEventListener("mouseup", onMouseUp);
      }

      document.addEventListener("mousemove", onMouseMove);
      document.addEventListener("mouseup", onMouseUp);
    }

    function startEncounter() {
      inBattle = true;
      document.getElementById("battle-screen").style.display = "block";
      document.getElementById("gameCanvas").style.display = "none";
    }

    function startBattle() {
      document.getElementById("battle-screen").style.display = "none";
      document.getElementById("gameCanvas").style.display = "block";
      inBattle = false;
    }

    drawPlayer();
  </script>
</body>
