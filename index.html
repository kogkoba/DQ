const SHEET_ID = "1GyKDfVQCNrBlkxjsrQDv_ouIio9yjO3mVQy6Ds5uQzg"; // Google Sheets の ID const API_KEY = "YOUR_GOOGLE_SHEETS_API_KEY"; // Google Cloud から取得した APIキー const SHEET_NAME = "クイズ"; // シート名 // Google Sheets からクイズデータを取得 async function fetchQuizData() { const url = `https://sheets.googleapis.com/v4/spreadsheets/${SHEET_ID}/values/${SHEET_NAME}?key=${API_KEY}`; const response = await fetch(url); const data = await response.json(); let quizList = []; let headers = data.values[0]; // ヘッダー let rows = data.values.slice(1); // データ部分 rows.forEach(row => { let quiz = {}; headers.forEach((header, index) => { quiz[header] = row[index] || ""; // 空のセルを考慮 }); quizList.push(quiz); }); return quizList; } // クイズの出題 (間違えが多い問題ほど出やすくする) async function getRandomQuiz() { let quizzes = await fetchQuizData(); let weightedQuizzes = []; quizzes.forEach(quiz => { let weight = parseInt(quiz["間違えた回数"]) + 1; // 1回も間違えてない問題も1回は出る for (let i = 0; i < weight; i++) { weightedQuizzes.push(quiz); } }); let selectedQuiz = weightedQuizzes[Math.floor(Math.random() * weightedQuizzes.length)]; return selectedQuiz; } // クイズを表示 async function startQuiz() { let quiz = await getRandomQuiz(); document.getElementById("question").textContent = quiz["問題文"]; document.getElementById("question-image").src = quiz["画像URL"]; let choicesDiv = document.getElementById("choices"); choicesDiv.innerHTML = ""; for (let i = 1; i <= 4; i++) { let choice = quiz[`選択肢${i}`]; let btn = document.createElement("button"); btn.textContent = choice; btn.className = "choice"; btn.onclick = () => checkAnswer(quiz, i); choicesDiv.appendChild(btn); } } // 解答チェック async function checkAnswer(quiz, selected) { let resultMessage = ""; if (parseInt(quiz["正解"]) === selected) { resultMessage = "✅ 正解！モンスターを倒した！"; } else { resultMessage = `❌ 不正解！ダメージを受けた！\n正解は「${quiz[`選択肢${quiz["正解"]}`]}」です。`; updateMistakeCount(quiz["ID"]); // 間違い回数を更新 } // 解説を表示 resultMessage += `\n\n🔎 解説: ${quiz["解説"]}`; alert(resultMessage); startQuiz(); // 次の問題を出題 } // 間違えた回数を Google Sheets に記録 async function updateMistakeCount(quizId) { const updateUrl = `https://sheets.googleapis.com/v4/spreadsheets/${SHEET_ID}/values/${SHEET_NAME}!J${parseInt(quizId) + 1}:J${parseInt(quizId) + 1}?valueInputOption=USER_ENTERED&key=${API_KEY}`; let body = { "range": `${SHEET_NAME}!J${parseInt(quizId) + 1}:J${parseInt(quizId) + 1}`, "values": [[parseInt(quizId["間違えた回数"]) + 1]] }; fetch(updateUrl, { method: "PUT", headers: { "Content-Type": "application/json" }, body: JSON.stringify(body) }).then(response => response.json()) .then(data => console.log("Updated mistake count", data)) .catch(error => console.error("Error updating mistake count", error)); } // 最初のクイズを出題 startQuiz();
